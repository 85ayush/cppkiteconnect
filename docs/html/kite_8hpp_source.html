986  njson _sendReq(
987  const http::method& mtd, const string& endpoint, const std::vector<std::pair<string, string>>& bodyParams = {}, bool isJson = false) {
988 
989  /*
990  If the endpoint expects pure JSON body, set isJson to true and put the json body in second element of bodyParam's first pair with first
991  element being empty string. see orderMargins() function
992  */
993 
994  // create request
995 
996  http::http_request req(mtd);
997  req.set_request_uri(U(endpoint));
998 
999  req.headers().set_content_type(U((isJson) ? "application/json" : "application/x-www-form-urlencoded"));
1000  req.headers().add(U("Authorization"), U(_getAuthStr()));
1001  req.headers().add(U("X-Kite-Version"), U(_kiteVersion));
1002 
1003  if ((mtd != http::methods::GET && mtd != http::methods::HEAD) && !bodyParams.empty()) {
1004 
1005  req.set_body(U((isJson) ? bodyParams[0].second : _encodeBody(bodyParams)));
1006  };
1007 
1008 
1009  // send request, get response and parse json
1010 
1011  http::http_response res = _httpClient.request(req).get();
1012  string dataRcvd = res.extract_string().get();
1013 
1014  if (!dataRcvd.empty()) {
1015 
1016  njson data;
1017 
1018  try {
1019 
1020  data = njson::parse(dataRcvd);
1021 
1022  } catch (const std::exception& e) { throw libException(FMT("{0} was thrown while parsing json (_sendReq-njson::parse)", e.what())); };
1023 
1024  if (res.status_code() == http::status_codes::OK) {
1025 
1026  // TODO can return njson["data"] & save users a step if it is established that every ok request has `data` field. It seems it is
1027  return data;
1028  }
1029 
1030  int code = 0;
1031  string excpStr;
1032  string message;
1033 
1034  try {
1035 
1036  code = res.status_code();
1037  excpStr = (data.contains("error_type")) ? data["error_type"] : "NoException";
1038  message = data["message"];
1039 
1040  } catch (const std::exception& e) {
1041 
1042  throw libException(
1043  FMT("{0} was thrown while extracting code({1}), excpStr({2}) and message({3}) (_sendReq)", e.what(), code, excpStr, message));
1044  };
1045 
1046  _throwException(excpStr, code, message);
1047 
1048 
1049  } else {
1050 
1051  return njson();
1052  };
1053 
1054  return njson();
1055  };
