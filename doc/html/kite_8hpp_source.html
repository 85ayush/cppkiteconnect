833  njson _sendReq(
834  const http::method& mtd, const string& endpoint, const std::vector<std::pair<string, string>>& bodyParams = {}, bool isJson = false) {
835 
836  /*
837  If the endpoint expects pure JSON body, set isJson to true and put the json body in second element of bodyParam's first pair with first
838  element being empty string. see orderMargins() function
839  */
840 
841  // create request
842 
843  http::http_request req(mtd);
844  req.set_request_uri(U(endpoint));
845 
846  req.headers().set_content_type(U((isJson) ? "application/json" : "application/x-www-form-urlencoded"));
847  req.headers().add(U("Authorization"), U(_getAuthStr()));
848  req.headers().add(U("X-Kite-Version"), U(_kiteVersion));
849 
850  if ((mtd != http::methods::GET && mtd != http::methods::HEAD) && !bodyParams.empty()) {
851 
852  req.set_body(U((isJson) ? bodyParams[0].second : _encodeBody(bodyParams)));
853  };
854 
855 
856  // send request, get response and parse json
857 
858  http::http_response res = _httpClient.request(req).get();
859  string dataRcvd = res.extract_string().get();
860 
861  if (!dataRcvd.empty()) {
862 
863  njson data;
864 
865  try {
866 
867  data = njson::parse(dataRcvd);
868 
869  } catch (const std::exception& e) { throw libException(FMT("{0} was thrown while parsing json (_sendReq-njson::parse)", e.what())); };
870 
871  if (res.status_code() == http::status_codes::OK) {
872 
873  // TODO can return njson["data"] & save users a step if it is established that every ok request has `data` field. It seems it is
874  return data;
875  }
876 
877  int code = 0;
878  string excpStr;
879  string message;
880 
881  try {
882 
883  code = res.status_code();
884  excpStr = (data.contains("error_type")) ? data["error_type"] : "NoException";
885  message = data["message"];
886 
887  } catch (const std::exception& e) {
888 
889  throw libException(
890  FMT("{0} was thrown while extracting code({1}), excpStr({2}) and message({3}) (_sendReq)", e.what(), code, excpStr, message));
891  };
892 
893  _throwException(excpStr, code, message);
894 
895 
896  } else {
897 
898  return njson();
899  };
900 
901  return njson();
902  };
