216 inline void _throwException(const string& excpStr, int code, const string& msg) {
217 
218 
219  // exception strings sent by API
220 
221  static const string tokenExcpStr = "TokenException";
222  static const string userExcpStr = "UserException";
223  static const string orderExcpStr = "OrderException";
224  static const string inputExcpStr = "InputException";
225  static const string networkExcpStr = "NetworkException";
226  static const string dataExcpStr = "DataException";
227  static const string generalExcpStr = "GeneralException";
228  static const string permissionExcpStr = "PermissionException";
229  static const string noExcpStr = "NoException"; // when REST API doesn't return any exception
230 
231  if (excpStr == tokenExcpStr) { throw tokenException(code, msg); };
232  if (excpStr == userExcpStr) { throw userException(code, msg); };
233  if (excpStr == orderExcpStr) { throw orderException(code, msg); };
234  if (excpStr == inputExcpStr) { throw inputException(code, msg); };
235  if (excpStr == networkExcpStr) { throw networkException(code, msg); };
236  if (excpStr == dataExcpStr) { throw dataException(code, msg); };
237  if (excpStr == generalExcpStr) { throw generalException(code, msg); };
238  if (excpStr == permissionExcpStr) { throw permissionException(code, msg); };
239  if (excpStr == noExcpStr) { throw noException(code, msg); };
240 
241  throw libException("Unknown exception was thrown by REST API");
242 };
